## Pages

<p>Number of Pages: {Object.keys(getPages()).length - 1}</p>
<ul>
    {Object.entries(getPages()).map(([key, page]) => {
        if (key.startsWith("/index")) {
            return;
        }
        return (
            <li key={key}>
                <a href={page.path} style={{textTransform: 'capitalize'}}>
                    {page.name}
                </a>
            </li>
        )
    })}
</ul>
